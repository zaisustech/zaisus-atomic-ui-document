Quick Start

Minimal examples to get each package running.

Atomic UI​

import React, { useState } from 'react';
import { View } from 'react-native';
import { Button, TextInputField, Card, Text } from '@zaisustech/atomic-ui';

export default function App() {
  const [email, setEmail] = useState('');

  return (
    <View style={{ padding: 20 }}>
      <Text variant="heading">Welcome</Text>
      <Card>
        <TextInputField
          label="Email"
          placeholder="Enter your email"
          value={email}
          onChangeText={setEmail}
        />
        <Button
          title="Submit"
          onPress={() => console.log('Submitted:', email)}
          variant="primary"
        />
      </Card>
    </View>
  );
}

Wrap your app with ThemeProvider (and optionally ToastProvider) from @zaisustech/atomic-ui when using components that need theme or toasts.

File System​

The file-system library provides a full UI (FileManagerContainer) and hooks for custom UIs.

Full file manager UI​

import { FileManagerContainer, FileManagerProvider } from '@zaisustech/file-system';
import { ThemeProvider, ToastProvider } from '@zaisustech/atomic-ui';
import { SafeAreaProvider } from 'react-native-safe-area-context';
import * as FileSystem from 'expo-file-system';

const documentPath = FileSystem.Paths.document.uri ?? '';

export default function App() {
  return (
    <SafeAreaProvider>
      <ThemeProvider>
        <ToastProvider>
          <FileManagerProvider>
            <FileManagerContainer
              path={documentPath}
              onSelect={(file) => {
                if (file.isDirectory) {
                  // Navigate to file.uri (e.g. push new screen)
                } else {
                  // Open or share file (e.g. expo-sharing)
                }
              }}
            />
          </FileManagerProvider>
        </ToastProvider>
      </ThemeProvider>
    </SafeAreaProvider>
  );
}

Custom list with hooks​

import { useFileManager, useFileActions, useFileSort } from '@zaisustech/file-system';
import * as FileSystem from 'expo-file-system';

const path = FileSystem.Paths.document.uri ?? '';

function CustomFileList() {
  const { files, isLoading, error, refresh } = useFileManager({ path });
  const { sortedFiles, sortKey, setSortKey, sortDirection, setSortDirection } = useFileSort({
    files,
    initialKey: 'name',
    initialDirection: 'asc',
  });
  const { copy, move, remove, rename, duplicate, createDirectory } = useFileActions({
    onRefresh: refresh,
  });

  // Render sortedFiles in your own FlatList; wire sortKey/setSortKey to your UI
  return (/* your list */);
}

Drawing (Signature pad)​

import React, { useRef } from 'react';
import { View, Button } from 'react-native';
import { SignaturePad, SignaturePadRef } from '@zaisustech/drawing';

export default function SignatureScreen() {
  const signatureRef = useRef<SignaturePadRef>(null);

  const handleSave = () => {
    signatureRef.current?.readSignature();
  };

  const handleClear = () => {
    signatureRef.current?.clearSignature();
  };

  const handleUndo = () => signatureRef.current?.undo();
  const handleRedo = () => signatureRef.current?.redo();

  return (
    <View style={{ flex: 1 }}>
      <SignaturePad
        ref={signatureRef}
        penColor="#000000"
        penSize={3}
        onSignatureOK={(dataUri) => {
          console.log('Signature captured:', dataUri);
          // Save or upload dataUri
        }}
      />
      <Button title="Undo" onPress={handleUndo} />
      <Button title="Redo" onPress={handleRedo} />
      <Button title="Save" onPress={handleSave} />
      <Button title="Clear" onPress={handleClear} />
    </View>
  );
}

See Drawing Overview and Drawing API for props and options.
